Example
var dancer = new Dancer();
var webvs = new Webvs.Main({
    canvas: document.getElementById("canvas"),
    analyser: new Webvs.DancerAdapter(dancer),
    showStat: true
});
webvs.loadPreset(samplePreset);
webvs.start();
dancer.load({src: "music.ogg"}); // start playing musc
dancer.play();
